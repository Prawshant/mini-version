 def takeInput(): 
     r = sr.Recognizer() 
     with sr.Microphone() as source: 
         print("Listining.........") 
         r.pause_threshold = 1 
         audio = r.listen(source) 
  
     try: 
         print("Rocognizing Voice........") 
         query = r.recognize_google(audio, language="en-in") 
         print(f"Sir you said : {query}\n") 
     except Exception as e: 
         print(e) 
         print("I didn't get it Sir.....") 
         return "None" 
     return query
